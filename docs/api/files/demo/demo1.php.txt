$score = new Score();

$measure = new Measure(
	array(
		'divisions' => 24,
	)
);
$note = new Note(
	array(
		'pitch' => 'C4',
		'duration' => 4,
		'type' => 'whole'
	)
);

$measure->addNote($note);
$part = new Part('Viola');
$part->addMeasure($measure);
$score->addPart($part);
$xml2 = $score->toXML('partwise');
